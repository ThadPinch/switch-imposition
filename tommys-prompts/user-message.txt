[Job.DataSet:Products]

[Job.DataSet:Kits]

    "externalOrderID": "<PO number>",
    "notes": "<general order notes>",
    "shippingMethod": "<shipping method or empty string>",
    "shippingAddresses": [
      {
        "company": "<company>",
        "attention": "<contact name>",
        "address1": "<address line 1>",
        "address2": "<address line 2 or empty>",
        "city": "<city>",
        "state": "<state abbrev>",
        "zip": "<zip>",
        "telephone": "<phone or empty>"
      }
    ],
    "orderItems": [
      {
        "productSKU": "<from Products catalog or empty>",

Output ONLY the JSON object.
